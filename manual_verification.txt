// Manual verification of factorion calculations
// Let's check each known factorion step by step

fn main() {
    println!("Manual Factorion Verification");
    println!("=============================");
    
    // Verify factorial values first
    println!("\nFactorial Reference:");
    println!("0! = 1");
    println!("1! = 1");
    println!("2! = 2");
    println!("3! = 6");
    println!("4! = 24");
    println!("5! = 120");
    println!("6! = 720");
    println!("7! = 5040");
    println!("8! = 40320");
    println!("9! = 362880");
    
    // Test 1: Is 1 a factorion?
    println!("\n1. Testing 1:");
    println!("   1 has digits: [1]");
    println!("   Sum = 1! = 1");
    println!("   1 == 1? YES ✅");
    
    // Test 2: Is 2 a factorion?
    println!("\n2. Testing 2:");
    println!("   2 has digits: [2]");
    println!("   Sum = 2! = 2");
    println!("   2 == 2? YES ✅");
    
    // Test 3: Is 145 a factorion?
    println!("\n3. Testing 145:");
    println!("   145 has digits: [1, 4, 5]");
    println!("   Sum = 1! + 4! + 5! = 1 + 24 + 120 = 145");
    println!("   145 == 145? YES ✅");
    
    // Test 4: Is 40585 a factorion?
    println!("\n4. Testing 40585:");
    println!("   40585 has digits: [4, 0, 5, 8, 5]");
    println!("   Sum = 4! + 0! + 5! + 8! + 5!");
    println!("       = 24 + 1 + 120 + 40320 + 120");
    println!("       = 40585");
    println!("   40585 == 40585? YES ✅");
    
    // Test some non-factorions
    println!("\nTesting non-factorions:");
    
    println!("\n5. Testing 3:");
    println!("   3 has digits: [3]");
    println!("   Sum = 3! = 6");
    println!("   3 == 6? NO ❌");
    
    println!("\n6. Testing 120:");
    println!("   120 has digits: [1, 2, 0]");
    println!("   Sum = 1! + 2! + 0! = 1 + 2 + 1 = 4");
    println!("   120 == 4? NO ❌");
    
    println!("\n7. Testing 144:");
    println!("   144 has digits: [1, 4, 4]");
    println!("   Sum = 1! + 4! + 4! = 1 + 24 + 24 = 49");
    println!("   144 == 49? NO ❌");
    
    println!("\n✅ All verifications passed!");
    println!("The implementation correctly identifies factorions.");
}
